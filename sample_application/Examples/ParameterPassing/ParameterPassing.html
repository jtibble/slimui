Parameter Passing

URL Parameter-passing can be used to select a particular item and drill down into its details.

To pass paramters to a view, you must add a list of parameters to the view in your config.json file: 

"ParameterPassing": {
	"url": "ParameterPassing",
	"template": "ParameterPassingTemplate",
	"model": "ParameterPassingModel",
	"modelbuilder": "ParameterPassingModelbuilder",
	"controller": "ParameterPassingController",
	"parameters": [
		"parameterExampleId",
		"parameterExampleId2"
	],
	"subroutes": [
		"ParameterPassingSubroute"	
	]
}
				
"ParameterPassingSubroute": {
	"template": "ParameterPassingSubrouteTemplate",
	"model": "ParameterPassingSubrouteModel",
	"modelbuilder": "ParameterPassingSubrouteModelbuilder",
	"controller": "ParameterPassingSubrouteController",
	"parameters": [
		"subrouteParameterExampleId",
		"subrouteParameterExampleId2"
	]
}

To receive parameters, include them in your controller and pass them to your modelbuilder: 

var ParameterPassingController = function( Context, Modelbuilder, StateParameters ){
	Context.Model = Modelbuilder( StateParameters );
};
				
var ParameterPassingModelbuilder = function(Model){
	return function( parameters ){
		
		if( parameters ){
			Model.parameters = parameters;
		}
		
		return Model;
	};
};
